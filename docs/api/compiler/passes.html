compiler.passes

Optimization and lowering passes for the DL compiler.

Each Pass transforms the IR in a specific, verifiable way. PassManager orchestrates execution with dependency tracking and optional verification between passes.

Passes: TilingPass — decompose large matmuls with cost model MemoryPlanningPass — liveness-aware buffer reuse GraphToSSAPass — lower Graph IR to SSA IR QuantizePass — FP32 → INT8 quantization

 49class PassManager:
 50    """Orchestrates ordered execution of compiler passes.
 51
 52    Usage::
 53
 54        pm = PassManager()
 55        pm.add(TilingPass(target_rows=4, target_cols=4))
 56        pm.add(MemoryPlanningPass())
 57        result = pm.run(ir)
 58        print(pm.summary())
 59    """
 60
 61    def __init__(self) -> None:
 62        self._passes: list[Pass] = []
 63        self.results: list[PassResult] = []
 64
 65    def add(self, p: Pass) -> None:
 66        self._passes.append(p)
 67
 68    def run(self, ir: Any) -> Any:
 69        """Execute all passes in order. Returns the final IR."""
 70        self.results = []
 71        current = ir
 72
 73        for p in self._passes:
 74            before = current
 75            current = p.run(current)
 76            warnings = p.verify(before, current)
 77            self.results.append(PassResult(
 78                pass_name=p.name,
 79                ir=current,
 80                warnings=warnings,
 81            ))
 82
 83        return current
 84
 85    def summary(self) -> str:
 86        """Return a human-readable summary of pass execution."""
 87        lines = ["Pass Pipeline Summary:", "-" * 40]
 88        for r in self.results:
 89            status = "✓" if not r.warnings else f"⚠ ({len(r.warnings)} warnings)"
 90            lines.append(f"  {r.pass_name:<30} {status}")
 91            for w in r.warnings:
 92                lines.append(f"    ⚠ {w}")
 93        return "\n".join(lines)

 67class TilingPass(Pass):
 68    """Decompose matmul nodes into tiled operations.
 69
 70    Searches candidate tile sizes using beam search, selecting
 71    the configuration with the minimum estimated SRAM misses.
 72
 73    Attributes:
 74        target_rows: Systolic array rows (default 4).
 75        target_cols: Systolic array columns (default 4).
 76        sram_capacity: SRAM capacity in 32-bit words (default 1024).
 77        beam_width: Number of candidates to keep per search step.
 78    """
 79
 80    name = "Tiling"
 81
 82    # Candidate tile sizes — powers of 2 from 2 to 128
 83    CANDIDATES = (2, 4, 8, 16, 32, 64, 128)
 84
 85    def __init__(
 86        self,
 87        target_rows: int = 4,
 88        target_cols: int = 4,
 89        sram_capacity: int = 1024,
 90        beam_width: int = 3,
 91    ) -> None:
 92        self.target_rows = target_rows
 93        self.target_cols = target_cols
 94        self.cost_model = CostModel(sram_capacity)
 95        self.beam_width = beam_width
 96
 97    def run(self, ir: Graph, **kwargs: Any) -> Graph:
 98        """Apply tiling to all matmul nodes in the graph.
 99
100        Each matmul/fused_mma_bias_relu node gets a 'tile' attribute
101        with the optimal (tm, tn, tk) configuration.
102        """
103        for node in ir.nodes:
104            if node.op in ("matmul", "fused_mma_bias_relu"):
105                shape = self._infer_shape(node)
106                if shape is None:
107                    continue
108                M, N, K = shape
109                best = self._search(M, N, K)
110                node.attrs["tile"] = (best.tm, best.tn, best.tk)
111                node.attrs["estimated_misses"] = best.estimated_misses
112
113        return ir
114
115    def verify(self, before: Graph, after: Graph) -> list[str]:
116        warnings = []
117        for node in after.nodes:
118            if node.op in ("matmul", "fused_mma_bias_relu"):
119                if "tile" not in node.attrs:
120                    warnings.append(
121                        f"{node.op} node {node.name or 'unnamed'} "
122                        f"missing tile config after TilingPass"
123                    )
124        return warnings

126    # ── Internal ─────────────────────────────────────────────────

128    def _search(self, M: int, N: int, K: int) -> TileConfig:
129        """Beam search over candidate tile sizes."""
130        candidates: list[TileConfig] = []
131
132        for tm in self.CANDIDATES:
133            if tm > M * 2:  # Don't tile much larger than the dimension
134                continue
135            for tn in self.CANDIDATES:
136                if tn > N * 2:
137                    continue
138                for tk in self.CANDIDATES:
139                    if tk > K * 2:
140                        continue
141                    misses = self.cost_model.estimate(M, N, K, tm, tn, tk)
142                    candidates.append(TileConfig(tm, tn, tk, misses))
143
144        if not candidates:
145            return TileConfig(
146                min(M, self.target_rows),
147                min(N, self.target_cols),
148                min(K, self.target_rows),
149            )
150
151        # Sort by misses, take top beam_width
152        candidates.sort(key=lambda c: c.estimated_misses)
153        return candidates[0]
154
155    @staticmethod
156    def _infer_shape(node: Node) -> tuple[int, int, int] | None:
157        """Infer (M, N, K) from node inputs.
158
159        Returns None if shapes cannot be determined.
160        """
161        import numpy as np
162
163        shapes: list[tuple[int, ...]] = []
164        for inp in node.inputs:
165            if isinstance(inp.value, np.ndarray):
166                shapes.append(inp.value.shape)
167            elif "shape" in inp.attrs:
168                shapes.append(inp.attrs["shape"])
169            elif inp.op == "input":
170                # Input node has no known shape — skip
171                shapes.append(())
172
173        if node.op == "fused_mma_bias_relu" and len(shapes) >= 2:
174            # x[M,K], w[K,N], b[N] — w is typically Const with known shape
175            # Find the weight (Const)
176            for i, inp in enumerate(node.inputs):
177                if inp.op == "const" and isinstance(inp.value, np.ndarray):
178                    w_shape = inp.value.shape
179                    if len(w_shape) == 2:
180                        K, N = w_shape
181                        # M is unknown for input, default to 4
182                        M = 4
183                        # Try to get M from x if it has shape info
184                        for j, other in enumerate(node.inputs):
185                            if j != i and isinstance(other.value, np.ndarray) and len(other.value.shape) == 2:
186                                M = other.value.shape[0]
187                        return (M, N, K)
188            return None
189
190        if node.op == "matmul" and len(shapes) == 2:
191            s0, s1 = shapes[0], shapes[1]
192            # If one operand has known shape, use it
193            if len(s0) == 2 and len(s1) == 2:
194                return (s0[0], s1[1], s0[1])
195            if len(s0) == 2:
196                # Only a has shape; assume b is Kx? with K=s0[1]
197                return (s0[0], s0[1], s0[1])  # N=K as fallback
198            if len(s1) == 2:
199                return (s1[1], s1[1], s1[0])  # M=K as fallback
200
201        return None

class MemoryPlanningPass(Pass): 
 34class MemoryPlanningPass(Pass):
 35    """Liveness-aware memory planner.
 36
 37    Reuses memory across non-overlapping buffer lifetimes to reduce
 38    peak allocation.
 39
 40    Attributes:
 41        alignment: Byte alignment for allocations (default 64 for cache line).
 42        element_bytes: Bytes per element for default type (default 4 for FP32).
 43    """
 44
 45    name = "Memory Planning"
 46
 47    def __init__(
 48        self,
 49        alignment: int = 64,
 50        element_bytes: int = 4,
 51    ) -> None:
 52        self.alignment = alignment
 53        self.element_bytes = element_bytes
 54
 55    def run(self, ir: Graph, **kwargs: Any) -> Graph:
 56        """Analyse liveness and assign buffer offsets.
 57
 58        Each node that produces a tensor gets ``buffer_offset`` and
 59        ``buffer_size`` attributes.  The total pool size is stored
 60        as ``total_pool_bytes`` in the graph's first node attrs
 61        (or via a synthetic marker).
 62        """
 63        if ir.output is None:
 64            return ir
 65
 66        # 1. Topological order (inputs first)
 67        try:
 68            order = ir.topological_order(ir.output)
 69        except RecursionError:
 70            return ir
 71
 72        # 2. Liveness: assign birth/death to each node's output
 73        step: dict[str, int] = {}
 74        for i, node in enumerate(order):
 75            step[id(node)] = i
 76
 77        buffers: list[BufferDesc] = []
 78        for i, node in enumerate(order):
 79            if node.op in ("input", "const"):
 80                continue  # input/const buffers managed externally
 81
 82            # Compute output size
 83            size = self._estimate_size(node)
 84            if size == 0:
 85                continue
 86
 87            # Death = max step of all consumers
 88            death = i
 89            for other in order:
 90                if node in other.inputs:
 91                    idx = step.get(id(other), i)
 92                    death = max(death, idx)
 93
 94            buffers.append(BufferDesc(
 95                name=node.name or f"{node.op}_{i}",
 96                size_bytes=size * self.element_bytes,
 97                birth=i,
 98                death=death,
 99                node=node,
100            ))
101
102        if not buffers:
103            return ir
104
105        # 3. Greedy allocation (first-fit by birth time)
106        buffers.sort(key=lambda b: b.birth)
107        active: list[BufferDesc] = []  # currently live buffers
108        free_list: list[tuple[int, int]] = [(0, 2 ** 30)]  # (base, limit)
109
110        for buf in buffers:
111            # GC dead buffers
112            active = [b for b in active if b.death >= buf.birth]
113            # Rebuild free list from active intervals
114            used = sorted((b.offset, b.offset + b.size_bytes) for b in active)
115            free_list = []
116            cursor = 0
117            for base, limit in used:
118                if cursor < base:
119                    free_list.append((cursor, base))
120                cursor = max(cursor, limit)
121            free_list.append((cursor, 2 ** 30))
122
123            # Allocate from first-fit free region
124            for base, limit in free_list:
125                if limit - base >= buf.size_bytes:
126                    aligned = (base + self.alignment - 1) // self.alignment * self.alignment
127                    if aligned + buf.size_bytes <= limit:
128                        buf.offset = aligned
129                        break
130
131            active.append(buf)
132
133            # Annotate the node
134            if buf.node:
135                buf.node.attrs["buffer_offset"] = buf.offset
136                buf.node.attrs["buffer_size"] = buf.size_bytes
137
138        # 4. Compute peak
139        if active:
140            peak = max(b.offset + b.size_bytes for b in active)
141            # Store in output node attrs
142            if ir.output:
143                ir.output.attrs["peak_pool_bytes"] = peak
144                ir.output.attrs["num_buffers"] = len(buffers)
145
146        return ir
147
148    def verify(self, before: Graph, after: Graph) -> list[str]:
149        warnings = []
150        if after.output and "peak_pool_bytes" in after.output.attrs:
151            peak = after.output.attrs["peak_pool_bytes"]
152            if peak > 16384:  # 16 KB warning
153                warnings.append(
154                    f"Peak memory {peak} bytes exceeds typical SRAM (16 KB)"
155                )
156        return warnings
157
158    # ── Internal ─────────────────────────────────────────────────
159
160    @staticmethod
161    def _estimate_size(node: Node) -> int:
162        """Estimate the number of elements in a node's output."""
163        if isinstance(node.value, np.ndarray):
164            return int(node.value.size)
165        if "shape" in node.attrs:
166            s = node.attrs["shape"]
167            n = 1
168            for d in s:
169                n *= d
170            return n
171        # Heuristic: default to 256 elements
172        return 256
