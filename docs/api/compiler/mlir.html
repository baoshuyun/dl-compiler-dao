212    """Lower tile operations to NPU dialect: LOAD/COMPUTE/STORE sequences.
213
214    Each tile becomes:
215        npu.load bank=0, addr=a_addr, size=a_size
216        npu.load bank=1, addr=b_addr, size=b_size
217        npu.barrier DMA
218        npu.compute a=0, b=1, c=2, dim=4x4, act=(RELU|NONE)
219        npu.barrier COMPUTE
220        npu.store bank=2, addr=c_addr, size=c_size
221        npu.barrier DMA
222    """
223
224    def run(self, tiles: list[TileOp],
225            a_base: int = 0, b_base: int = 64, c_base: int = 128) -> _LowerResult:
226        """Lower tiles to NPU ops with correct 2D→1D address mapping.
227
228        Returns (_LowerResult):
229            ops: list of NpuOp
230            c_addrs: dict[c_off_tuple] = list of external memory addresses
231                     where partial K-tile results were stored.
232                     Caller must accumulate across K-tiles in the same group.
233        """
234        ops = []
235        c_partials: dict[tuple[int, int], list[int]] = {}
236        c_next = c_base
237
238        for i, tile in enumerate(tiles):
239            # Fixed bank allocation: A=0, B=1, C=2
240            # No double-buffering — correctness first, speed later
241            a_bank, b_bank, c_bank = 0, 1, 2
242
243            # Load A tile rows
244            if tile.a_contiguous:
245                addr = tile.a_row_addr(0, a_base)
246                ops.append(NpuLoad(bank=a_bank, base_addr=addr, size=tile.load_a_size))
247            else:
248                for r in range(tile.m):
249                    addr = tile.a_row_addr(r, a_base)
250                    ops.append(NpuLoad(bank=a_bank, base_addr=addr, size=tile.k))
251
252            # Load B tile rows
253            if tile.b_contiguous:
254                addr = tile.b_row_addr(0, b_base)
255                ops.append(NpuLoad(bank=b_bank, base_addr=addr, size=tile.load_b_size))
256            else:
257                for r in range(tile.k):
258                    addr = tile.b_row_addr(r, b_base)
259                    ops.append(NpuLoad(bank=b_bank, base_addr=addr, size=tile.n))
260
261            ops.append(NpuBarrier(BarrierType.DMA))
262
263            # Each K-tile gets a unique C output address for partial sum accumulation
264            tile_c_addr = c_next
265            c_next += tile.store_c_size
266            key = (tile.c_off[0], tile.c_off[1])
267            c_partials.setdefault(key, []).append(tile_c_addr)
268
269            ops.append(NpuCompute(
270                a_src=a_bank, b_src=b_bank, c_dst=c_bank,
271                dim=MatDim.MAT_4x4, act=tile.act,
272            ))
273            ops.append(NpuBarrier(BarrierType.COMPUTE))
274

276                ops.append(NpuStore(bank=c_bank, base_addr=tile_c_addr, size=tile.store_c_size))
277            else:
278                store_addr = tile_c_addr
279                for r in range(tile.m):
280                    ops.append(NpuStore(bank=c_bank, base_addr=store_addr, size=tile.n))
281                    store_addr += tile.n
282
283            ops.append(NpuBarrier(BarrierType.DMA))
284
285        return _LowerResult(ops, c_partials)
