compiler.isa_mapper

ISA Mapper — compiler IR to NPU hardware instruction bridge.

The ISA Mapper is the backend that translates optimized deep-learning compiler IR into a stream of 32-bit NPU instructions. It is the critical missing link between the software compiler (mini-dl-compiler) and the hardware accelerator (NPU_Project).

Pipeline::

Lowered IR ops
    │
    ▼
Tiler           — decompose large matmuls into 4×4 NPU tiles
    │
    ▼
Scheduler       — linearise, insert BARRIERs for sync
    │
    ▼
BankAllocator   — assign tensors to SRAM banks (ping/pong)
    │
    ▼
InstructionEmitter — TileOp / DMA → NPUInstruction
    │
    ▼
Assembler       — encode to 32-bit binary instr_mem
    │
    ▼
Simulator       — run on Python golden model, compare results

 76    """Top-level ISA Mapper: lowers compiler IR ops to an NPU program.
 77
 78    This is the main entry point.  It orchestrates tiling, scheduling,
 79    bank allocation, instruction emission, and assembly.
 80
 81    Usage::
 82
 83        mapper = ISAMapper()
 84        program = mapper.map_mlp(
 85            M=128, N=128, K=128,
 86            ext_a_addr=0, ext_b_addr=64, ext_c_addr=128,
 87        )

 89
 90        # Run on the simulator to verify correctness
 91        sim = NPUSimulator()
 92        sim.load_program(program)
 93        sim.load_external_memory({i: 2 for i in range(256)})
 94        sim.run()
 95        results = sim.read_external_memory(128, 16)

108    def map_matmul(
109        self,
110        M: int,
111        N: int,
112        K: int,
113        *,
114        ext_a_addr: int = 0,
115        ext_b_addr: int = 64,
116        ext_c_addr: int = 128,
117        act: ActType = ActType.NONE,
118    ) -> NPUProgram:
119        """Map a single matmul C[M,N] = A[M,K] @ B[K,N] to NPU instructions.
120
121        For multi-tile matmuls (M>4 or N>4 or K>4), each tile reloads its
122        sub-matrix from external DRAM before computing, since the current
123        COMPUTE ISA has no SRAM offset field.
124
125        Args:
126            M, N, K: Matrix dimensions.
127            ext_a_addr: External DRAM base address for matrix A.
128            ext_b_addr: External DRAM base address for matrix B.
129            ext_c_addr: External DRAM base address for result C.
130            act: Optional activation function.
131
132        Returns:
133            An assembled NPUProgram ready for execution.
134        """
135        a_size = M * K
136        b_size = K * N
137        c_size = M * N
138
139        # Step 1: Tile
140        tiles = self.tiler.tile_matmul("A", "B", "C", M, N, K, act=act)
141        tm = self.config.array_rows
142        tn = self.config.array_cols
143
144        # Step 2: Build per-tile DMA + compute schedule
145        # Each tile: LOAD A_sub → LOAD B_sub → COMPUTE → STORE C_sub
146        # This works because each tile writes a disjoint C region
147        # (K-tiling not yet supported — requires accumulate mode)
148        all_ops: list[dict[str, Any] | TileOp] = []
149
150        for tile in tiles:
151            mi, mj, mk = tile.tile_idx
152            m_start = tile.c_slice[0].start
153            n_start = tile.c_slice[1].start
154            m_size = tile.c_slice[0].stop - m_start
155            n_size = tile.c_slice[1].stop - n_start
156            k_size = tile.a_slice[1].stop - tile.a_slice[1].start
157
158            a_start = tile.a_slice[0].start * K + tile.a_slice[1].start
159            a_count = m_size * k_size
160            b_start = tile.b_slice[0].start * N + tile.b_slice[1].start
161            b_count = k_size * n_size
162            c_start = ext_c_addr + m_start * N + n_start
163            c_count = m_size * n_size
164
165            tile_key = f"t{mi}_{mj}_{mk}"
166            tile_a_name = f"A_{tile_key}"
167            tile_b_name = f"B_{tile_key}"
168            tile_c_name = f"C_{tile_key}"
169
170            all_ops.append({
171                "type": "load", "buf": tile_a_name,
172                "bank": 0, "ext_addr": ext_a_addr + a_start, "size": a_count,
173            })
174            all_ops.append({
175                "type": "load", "buf": tile_b_name,
176                "bank": 1, "ext_addr": ext_b_addr + b_start, "size": b_count,
177            })
178            all_ops.append(TileOp(
179                op_type="matmul",
180                a_name=tile_a_name,
181                b_name=tile_b_name,
182                c_name=tile_c_name,
183                a_slice=tile.a_slice,
184                b_slice=tile.b_slice,
185                c_slice=tile.c_slice,
186                mat_dim=tile.mat_dim,
187                act=tile.act,
188                tile_idx=tile.tile_idx,
189            ))
190            all_ops.append({
191                "type": "store", "buf": tile_c_name,
192                "bank": 2, "ext_addr": c_start, "size": c_count,
193            })
194
195        # Step 3: Schedule with interleaved per-tile ordering
196        scheduled = self.scheduler.schedule([], interleaved=all_ops)
197
198        # Step 4: Bank allocation — simple: A→0, B→1, C→2
199        live_ranges = self._build_live_ranges_multi(scheduled, a_size, b_size, c_size, tiles)
200        assignments = self.bank_allocator.allocate(live_ranges)
201
202        # Step 5: Update DMA op bank numbers from actual assignments
203        for sop in scheduled:
204            if sop.op == "dma" and sop.dma_info:
205                buf_name = sop.dma_info["buf"]
206                if buf_name in assignments:
207                    sop.dma_info["bank"] = assignments[buf_name].bank
208
209        # Step 6: Emit instructions
210        self.emitter.bank_assignments = assignments
211        instructions = self.emitter.emit(scheduled)
212
213        # Step 7: Assemble
214        return self.assembler.assemble(instructions)

242    def verify(
243        self,
244        program: NPUProgram,
245        inputs: dict[str, NDArray[np.int32]],
246    ) -> dict[str, NDArray[np.int32]]:
247        """Run *program* on the NPU simulator and return results.
248
249        Args:
250            program: Assembled NPU program.
251            inputs: Dict mapping buffer name → NumPy array.
252                    Supported keys: "A", "B", "X", "W", "b".
253
254        Returns:
255            Dict with "C" → result array.
256        """
257        sim = NPUSimulator(self.config)
258
259        # Load input data into external memory
260        ext_mem: dict[int, int] = {}
261        offset = 0
262        for key, arr in inputs.items():
263            flat = arr.ravel().astype(np.int32)
264            for i, val in enumerate(flat):
265                ext_mem[offset + i] = int(val)
266            offset += len(flat)
267
268        sim.load_program(program)
269        sim.load_external_memory(ext_mem)
270        sim.run()
271
272        # Determine output size and read
273        c_size = 0
274        for instr in program.instructions:
275            if instr._opcode_int == 0x2:  # STORE
276                c_size = (instr.binary >> 16) & 0x3FF
277                break
278
279        if c_size == 0:
280            c_size = 16  # default 4x4
281
282        results = sim.read_external_memory(offset, c_size)
283        return {"C": np.array(results, dtype=np.int32)}

287    @staticmethod
288    def _build_live_ranges_multi(
289        scheduled: list[ScheduledOp],
290        a_size: int,
291        b_size: int,
292        c_size: int,
293        tiles: list[TileOp],
294    ) -> list[_LiveRange]:
295        """Extract liveness information from a multi-tile schedule.
296
297        Each per-tile buffer is live from its first use (load or compute)
298        to its last use (compute read or store).
299        """
300        live: list[_LiveRange] = []
301        max_step = len(scheduled)
302
303        # Collect all buffer names and their first/last use from the schedule
304        buf_first: dict[str, int] = {}
305        buf_last: dict[str, int] = {}
306        buf_sizes: dict[str, int] = {}
307
308        for i, sop in enumerate(scheduled):
309            if isinstance(sop.op, TileOp):
310                t = sop.op
311                for name in (t.a_name, t.b_name):
312                    if name not in buf_first:
313                        buf_first[name] = i
314                    buf_last[name] = i
315                    m_sz = t.c_slice[0].stop - t.c_slice[0].start
316                    n_sz = t.c_slice[1].stop - t.c_slice[1].start
317                    k_sz = t.a_slice[1].stop - t.a_slice[1].start
318                    buf_sizes[name] = max(buf_sizes.get(name, 0),
319                                          m_sz * k_sz if "A" in name else k_sz * n_sz)
320                c_name = t.c_name
321                if c_name not in buf_first:
322                    buf_first[c_name] = i
323                buf_last[c_name] = i
324                buf_sizes[c_name] = max(buf_sizes.get(c_name, 0),
325                                        m_sz * n_sz)
326            elif sop.op == "dma" and sop.dma_info:
327                buf_name = sop.dma_info["buf"]
328                if buf_name not in buf_first:
329                    buf_first[buf_name] = i
330                buf_last[buf_name] = i + 1
331
332        for name in buf_first:
333            is_out = name.startswith("C_") or name == "C"
334            live.append(_LiveRange(
335                name=name,
336                num_words=buf_sizes.get(name, 16),
337                first_use=buf_first[name],
338                last_use=buf_last[name],
339                is_input=name.startswith("A_") or name.startswith("B_"),
340                is_output=is_out,
341            ))
342
343        return live

 48    def allocate(
 49        self,
 50        live_ranges: list[_LiveRange],
 51    ) -> dict[str, BufferAssignment]:
 52        """Assign each live range to a bank and offset.
 53
 54        Strategy:
 55          1. Input buffers A, B → banks 0, 1 (ping).
 56          2. Output buffers → banks 2, 3 (or reuse input banks after last use).
 57          3. Intermediate buffers → first-fit across all 4 banks.
 58          4. Use liveness to reuse space: a dead buffer's space can be
 59             given to a new buffer.
 60
 61        Returns:
 62            Dict mapping buffer name → BufferAssignment.
 63        """
 64        # Per-bank free list: list of (base, limit) free regions
 65        banks: list[list[tuple[int, int]]] = [
 66            [(0, self.bank_depth)] for _ in range(self.config.sram_banks)
 67        ]
 68
 69        # Per-bank active allocations: [(name, base, limit, last_use)]
 70        active: list[list[tuple[str, int, int, int]]] = [[] for _ in range(self.config.sram_banks)]
 71
 72        assignments: dict[str, BufferAssignment] = {}
 73
 74        for buf in sorted(live_ranges, key=lambda b: b.first_use):
 75            # 1. Garbage-collect dead buffers before allocating
 76            for bank_idx in range(self.config.sram_banks):
 77                self._gc(bank_idx, buf.first_use, banks, active)
 78
 79            # 2. Pick bank
 80            preferred = self._preferred_banks(buf)
 81            bank_idx = self._find_bank(buf.num_words, banks, preferred)
 82
 83            if bank_idx is None:
 84                # Fallback: try any bank
 85                bank_idx = self._find_bank(buf.num_words, banks, None)
 86
 87            if bank_idx is None:
 88                raise RuntimeError(

 90                    f"all banks full at step {buf.first_use}"
 91                )
 92
 93            # 3. Allocate from the chosen bank
 94            base = self._alloc_from_bank(buf.num_words, banks[bank_idx])

 96                name=buf.name,
 97                bank=bank_idx,

108    def _preferred_banks(buf: _LiveRange) -> list[int] | None:
109        """Return preferred bank indices for a buffer, or None for any."""
110        name_lower = buf.name.lower()
111        if "a" == name_lower or name_lower.startswith("a_") or name_lower.startswith("input_a"):
112            return [0, 2]  # ping/pong for A
113        if "b" == name_lower or name_lower.startswith("b_") or name_lower.startswith("input_b"):
114            return [1, 3]  # ping/pong for B
115        if buf.is_output:
116            return [2, 3]  # output banks

120    def _find_bank(
121        num_words: int,
122        banks: list[list[tuple[int, int]]],
123        preferred: list[int] | None,
124    ) -> int | None:
125        """Find the first bank (preferring *preferred*) with a free region ≥ num_words."""
126        candidates = preferred if preferred is not None else list(range(len(banks)))
127        for bank_idx in candidates:
128            for base, limit in banks[bank_idx]:
129                if limit - base >= num_words:

134    def _alloc_from_bank(num_words: int, free_list: list[tuple[int, int]]) -> int:
135        """Allocate *num_words* from the first-fitting free region. Returns base address."""
136        for i, (base, limit) in enumerate(free_list):
137            if limit - base >= num_words:
138                free_list[i] = (base + num_words, limit)
139                if free_list[i][0] >= free_list[i][1]:
140                    free_list.pop(i)
141                return base
142        raise RuntimeError("No free region found (inconsistent state)")

150    ) -> None:
151        """Reclaim space from buffers whose last_use < current_step."""
152        new_active = []

155            if last_use < current_step:

157            else:

168                merged[-1] = (merged[-1][0], limit)
169            else:
170                merged.append((base, limit))

265        # A: rows × k_dim, stored row-major
266        A = np.zeros((rows, k_dim), dtype=np.int32)
267        for r in range(rows):
268            for k in range(k_dim):
269                idx = r * k_dim + k
270                if idx < self.bank_depth:
271                    A[r, k] = a_bank[idx]
272
273        # B: k_dim × cols, stored row-major
274        B = np.zeros((k_dim, cols), dtype=np.int32)
275        for k in range(k_dim):
276            for c in range(cols):
277                idx = k * cols + c
278                if idx < self.bank_depth:
279                    B[k, c] = b_bank[idx]
280
281        # C = A @ B
282        C = A.astype(np.int64) @ B.astype(np.int64)

284        # Activation

290            for c in range(cols):

292                if idx < self.bank_depth:
