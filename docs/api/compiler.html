174    def reachable_from(self, root: Node) -> set[Node]:
175        """Return the set of nodes reachable from *root* (including *root*)."""

188    def topological_order(self, root: Node) -> list[Node]:
189        """Return nodes reachable from *root* in topological order (inputs first)."""
190        order: list[Node] = []
191        perm: set[Node] = set()
192        temp: set[Node] = set()

194        def visit(n: Node) -> None:
195            if n in perm:
196                return

198                raise RecursionError("Graph contains a cycle")

202            temp.discard(n)
203            perm.add(n)
204            order.append(n)

 15class NumpyCompiler:
 16    """Compiles an optimized computation graph into a Python function backed by NumPy."""
 17
 18    def __init__(self) -> None:
 19        self.lines: list[str] = []
 20        self._memo: dict[Node, str] = {}
 21        self._tmp_counter = 0
 22
 23    # ------------------------------------------------------------------
 24    # Public API
 25    # ------------------------------------------------------------------
 26
 27    def compile(
 28        self,
 29        output: Node,
 30        *,
 31        graph: Graph | None = None,
 32        optimize: bool = True,
 33    ) -> tuple[Callable[..., Any], str, Node, list[str]]:
 34        """Compile *output* (and its dependencies) into a callable.
 35
 36        Args:
 37            output: Root node of the computation.
 38            graph: Optional graph (used for DCE when *optimize* is True).
 39            optimize: Run the standard optimization pipeline before codegen.
 40
 41        Returns:
 42            (callable, source_code, optimized_output_node, optimization_logs)
 43        """
 44        logs: list[str] = []
 45
 46        if optimize and graph is not None:
 47            opt = Optimizer(graph)
 48            graph.output = output
 49            output = opt.run()
 50            logs = opt.logs
 51
 52        args = self._collect_inputs(output)
 53        self._reset()
 54
 55        self.lines = [
 56            "import numpy as np",
 57            "",
 58            f"def compiled({', '.join(args)}):",
 59        ]
 60
 61        out_var = self._emit(output)
 62        self.lines.append(f"    return {out_var}")
 63
 64        source = "\n".join(self.lines)
 65
 66        namespace: dict[str, Any] = {}
 67        exec(source, namespace)
 68        return namespace["compiled"], source, output, logs
 69
 70    def compile_to_source(self, output: Node, *, optimize: bool = True) -> str:
 71        """Return only the generated source code (no exec)."""
 72        _, source, _, _ = self.compile(output, optimize=optimize)
 73        return source

 75    # ------------------------------------------------------------------
 76    # Internals
 77    # ------------------------------------------------------------------
 78
 79    def _reset(self) -> None:
 80        self._memo.clear()
 81        self._tmp_counter = 0
 82
 83    def _new_tmp(self) -> str:
 84        self._tmp_counter += 1
 85        return f"v{self._tmp_counter}"
 86
 87    @staticmethod
 88    def _encode(value: np.ndarray[Any, Any]) -> str:
 89        return f"np.array({value.tolist()!r}, dtype=np.{value.dtype.name})"
 90
 91    def _collect_inputs(self, root: Node) -> list[str]:
 92        ordered: list[str] = []
 93        seen_nodes: set[Node] = set()
 94        seen_names: set[str] = set()
 95
 96        def visit(n: Node) -> None:
 97            if n in seen_nodes:
 98                return
 99            seen_nodes.add(n)
100            if n.op == "input" and n.name not in seen_names:
101                ordered.append(n.name)
102                seen_names.add(n.name)
103            for child in n.inputs:
104                visit(child)
105
106        visit(root)
107        return ordered
108
109    def _emit(self, node: Node) -> str:
110        """Emit code for *node* and return the variable name holding its value."""
111        if node in self._memo:
112            return self._memo[node]
113
114        if node.op == "input":
115            self._memo[node] = node.name
116            return node.name
117
118        if node.op == "const":
119            var = self._new_tmp()
120            self.lines.append(f"    {var} = {self._encode(node.value)}")
121            self._memo[node] = var
122            return var
123
124        # Gather operand variable names
125        operands = [self._emit(inp) for inp in node.inputs]
126
127        var = self._new_tmp()
128
129        if node.op == "add":
130            self.lines.append(f"    {var} = {operands[0]} + {operands[1]}")
131        elif node.op == "matmul":
132            self.lines.append(f"    {var} = {operands[0]} @ {operands[1]}")
133        elif node.op == "relu":
134            self.lines.append(f"    {var} = np.maximum({operands[0]}, 0)")
135        elif node.op == "fused_mma_bias_relu":
136            x, w, b = operands
137            self.lines.append(f"    {var} = np.maximum(({x} @ {w}) + {b}, 0)")
138        elif node.op == "conv2d":
139            x, w = operands
140            strides = node.attrs.get("strides", (1, 1))
141            padding = node.attrs.get("padding", (0, 0))
142            self.lines.extend(self._emit_conv2d(var, x, w, strides, padding))
143        elif node.op == "max_pool2d":
144            ks = node.attrs.get("kernel_size", (2, 2))
145            st = node.attrs.get("strides", ks)
146            self.lines.extend(self._emit_maxpool2d(var, operands[0], ks, st))
147        elif node.op == "batch_norm":
148            x, s, b, m, v = operands
149            eps = node.attrs.get("epsilon", 1e-5)
150            self.lines.append(
151                f"    {var} = ({x} - {m}) / np.sqrt({v} + {eps}) * {s} + {b}")
152        elif node.op == "fused_bn_relu":
153            x, s, b, m, v = operands
154            eps = node.attrs.get("epsilon", 1e-5)
155            self.lines.append(
156                f"    {var} = np.maximum(({x} - {m}) / np.sqrt({v} + {eps}) * {s} + {b}, 0)")
157        elif node.op == "softmax":
158            axis = node.attrs.get("axis", -1)
159            self.lines.append(
160                f"    {var} = np.exp({operands[0]} - {operands[0]}.max(axis={axis}, keepdims=True))")
161            self.lines.append(
162                f"    {var} = {var} / {var}.sum(axis={axis}, keepdims=True)")
163        elif node.op == "reshape":
164            shape = node.attrs["shape"]
165            self.lines.append(f"    {var} = {operands[0]}.reshape({shape})")
166        elif node.op == "transpose":
167            axes = node.attrs.get("axes", None)
168            axes_str = str(axes) if axes else ""
169            self.lines.append(f"    {var} = {operands[0]}.transpose({axes_str})" if axes else
170                            f"    {var} = {operands[0]}.T")
171        elif node.op == "fused_conv_bias_relu":
172            x, w, b = operands[:3]
173            strides = node.attrs.get("strides", (1, 1))
174            padding = node.attrs.get("padding", (0, 0))
175            tmp = self._new_tmp()
176            self.lines.extend(self._emit_conv2d(tmp, x, w, strides, padding))
177            self.lines.append(f"    {var} = np.maximum({tmp} + {b}, 0)")
178        else:
179            raise ValueError(f"Unknown op: {node.op!r}")
180
181        self._memo[node] = var
182        return var

186    @staticmethod
187    def _emit_conv2d(var: str, x: str, w: str,
188                     strides: tuple[int, int],
189                     padding: tuple[int, int]) -> list[str]:
190        """Emit im2col-based conv2d NumPy code.
191
192        Returns lines to append (caller adds to self.lines).
193        """
194        # Simple implementation using nested loops for clarity.
195        # In production, use np.lib.stride_tricks.as_strided + tensordot.
196        ph, pw = padding
197        sh, sw = strides
198        lines = [
199            f"    # conv2d: {x} @ {w} strides={strides} pad={padding}",
200            f"    _x = {x}",
201            f"    _w = {w}",
202        ]
203        if ph > 0 or pw > 0:
204            lines.append(
205                f"    _x = np.pad(_x, ((0,0),({ph},{ph}),({pw},{pw}),(0,0)))")
206        lines.extend([
207            f"    _n, _h, _w_in, _c_in = _x.shape",
208            f"    _kh, _kw, _c_in2, _c_out = _w.shape",
209            f"    _oh = (_h - _kh) // {sh} + 1",
210            f"    _ow = (_w_in - _kw) // {sw} + 1",
211            f"    # im2col + matmul",

 48    def fold(self, node: Node, memo: dict[Node, Node] | None = None) -> Node:
 49        """Bottom-up constant folding.
 50
 51        Returns a new (or existing) node that is the folded equivalent
 52        of *node*.  Pure-constant sub-graphs are evaluated at compile time.
 53        """
 54        if memo is None:
 55            memo = {}
 56
 57        if node in memo:
 58            return memo[node]
 59
 60        # Leaf nodes don't change.
 61        if node.op in ("input", "const"):
 62            memo[node] = node
 63            return node
 64
 65        # Recurse on inputs first.
 66        folded_inputs = [self.fold(inp, memo) for inp in node.inputs]
 67
 68        out: Node
 69
 70        if node.op == "add":
 71            a, b = folded_inputs
 72
 73            # Identity: x + 0 == 0 + x == x
 74            if a.op == "const" and np.all(a.value == 0):
 75                self._log(f"[fold] add(0, x) -> x  ({_label(b)})")
 76                out = b
 77            elif b.op == "const" and np.all(b.value == 0):
 78                self._log(f"[fold] add(x, 0) -> x  ({_label(a)})")
 79                out = a
 80            elif a.op == "const" and b.op == "const":
 81                self._log(
 82                    f"[fold] const + const -> const  "
 83                    f"({a.value.shape} + {b.value.shape})"
 84                )
 85                out = self.graph.Const(a.value + b.value)
 86            elif folded_inputs == node.inputs:
 87                out = node  # no change — reuse the original
 88            else:
 89                out = Node(op="add", inputs=folded_inputs)
 90                self._copy_meta(node, out)
 91
 92        elif node.op == "matmul":
 93            a, b = folded_inputs
 94
 95            if a.op == "const" and b.op == "const":
 96                self._log(
 97                    f"[fold] const @ const -> const  "
 98                    f"({a.value.shape} @ {b.value.shape})"
 99                )
100                out = self.graph.Const(a.value @ b.value)
101            elif folded_inputs == node.inputs:
102                out = node
103            else:
104                out = Node(op="matmul", inputs=folded_inputs)
105                self._copy_meta(node, out)
106
107        elif node.op == "relu":
108            a = folded_inputs[0]
109
110            if a.op == "const":
111                self._log(
112                    f"[fold] relu(const) -> const  ({a.value.shape})"
113                )
114                out = self.graph.Const(np.maximum(a.value, 0))
115            elif folded_inputs[0] is node.inputs[0]:
116                out = node
117            else:
118                out = Node(op="relu", inputs=folded_inputs)
119                self._copy_meta(node, out)
120
121        elif node.op == "fused_mma_bias_relu":
122            x, w, b = folded_inputs
123            if x.op == "const" and w.op == "const" and b.op == "const":
124                self._log(
125                    f"[fold] fused_mma_bias_relu(all-const) -> const  "
126                    f"({x.value.shape} @ {w.value.shape} + {b.value.shape})"
127                )
128                out = self.graph.Const(np.maximum((x.value @ w.value) + b.value, 0))
129            elif folded_inputs == node.inputs:
130                out = node
131            else:
132                out = Node(op="fused_mma_bias_relu", inputs=folded_inputs)
133                self._copy_meta(node, out)
134
135        elif node.op in ("conv2d", "max_pool2d", "batch_norm",
136                         "softmax", "reshape", "transpose"):
137            # Element-wise passthrough for ops without const-folding (yet)
138            if folded_inputs == node.inputs:
139                out = node
140            else:
141                out = Node(op=node.op, inputs=folded_inputs,
142                          attrs=dict(node.attrs))
143                self._copy_meta(node, out)
144
145        else:
146            raise ValueError(f"Unknown op: {node.op!r}")
147
148        memo[node] = out
149        return out

155    def fuse(self, node: Node, memo: dict[Node, Node] | None = None) -> Node:
156        """Bottom-up operator fusion.
157
158        Merges ``matmul -> add -> relu`` patterns into a single
159        ``fused_mma_bias_relu`` node.
160        """
161        if memo is None:
162            memo = {}
163
164        if node in memo:
165            return memo[node]
166
167        if node.op in ("input", "const"):
168            memo[node] = node
169            return node
170
171        fused_inputs = [self.fuse(inp, memo) for inp in node.inputs]
172
173        out: Node
174
175        if node.op == "relu":
176            child = fused_inputs[0]
177            if child.op == "add":
178                left, right = child.inputs
179                # Try both orderings: matmul + bias  OR  bias + matmul
180                if left.op == "matmul":
181                    self._log("[fuse] matmul + add + relu -> fused_mma_bias_relu")
182                    out = Node(
183                        op="fused_mma_bias_relu",
184                        inputs=[left.inputs[0], left.inputs[1], right],
185                    )
186                    self._copy_meta(node, out)
187                    memo[node] = out
188                    return out
189                if right.op == "matmul":
190                    self._log("[fuse] matmul + add + relu -> fused_mma_bias_relu")
191                    out = Node(
192                        op="fused_mma_bias_relu",
193                        inputs=[right.inputs[0], right.inputs[1], left],
194                    )
195                    self._copy_meta(node, out)
196                    memo[node] = out
197                    return out
198
199        elif node.op == "add":
200            # Conv2D + add + relu fusion: Add(Conv2D(x,w), b) → FusedConvBias
201            child_left, child_right = fused_inputs
202            if child_left.op == "conv2d" and child_right.op == "const":
203                self._log("[fuse] conv2d + add -> fused_conv_bias")
204                out = Node(op="fused_conv_bias_relu",
205                          inputs=child_left.inputs + [child_right],
206                          attrs=dict(child_left.attrs))
207                self._copy_meta(node, out)
208                memo[node] = out
209                return out
210            if child_right.op == "conv2d" and child_left.op == "const":
211                self._log("[fuse] conv2d + add -> fused_conv_bias")
212                out = Node(op="fused_conv_bias_relu",
213                          inputs=child_right.inputs + [child_left],
214                          attrs=dict(child_right.attrs))
215                self._copy_meta(node, out)
216                memo[node] = out
217                return out
218
219        elif node.op == "relu":
220            child = fused_inputs[0]
221            # FusedConvBias + relu → fused_conv_bias_relu
222            if child.op == "fused_conv_bias":
223                self._log("[fuse] fused_conv_bias + relu -> fused_conv_bias_relu")
224                out = Node(op="fused_conv_bias_relu",
225                          inputs=child.inputs,
226                          attrs=dict(child.attrs))
227                self._copy_meta(node, out)
228                memo[node] = out
229                return out
230            # BatchNorm + relu fusion
231            if child.op == "batch_norm":
232                self._log("[fuse] batch_norm + relu -> fused_bn_relu")
233                out = Node(op="fused_bn_relu",
234                          inputs=child.inputs,
235                          attrs=dict(child.attrs))
236                self._copy_meta(node, out)
237                memo[node] = out
238                return out
239
240        if fused_inputs == node.inputs:
241            out = node
242        else:
243            out = Node(op=node.op, inputs=fused_inputs, attrs=dict(node.attrs))
244            self._copy_meta(node, out)
245
246        memo[node] = out
247        return out

253    def dce(self, output: Node) -> Node:
254        """Remove nodes from ``graph.nodes`` that are not reachable from *output*.
255
256        Returns *output* unchanged (the graph topology stays the same; only the
257        node list is trimmed).
258        """
259        reachable = self.graph.reachable_from(output)
260        removed = [n for n in self.graph.nodes if n not in reachable]
261
262        if removed:
263            names = []
264            for n in removed[:8]:
265                names.append(n.name or n.op)
266            suffix = f" ... (+{len(removed) - 8})" if len(removed) > 8 else ""
267            self._log(f"[dce] removed {len(removed)} dead node(s): {', '.join(names)}{suffix}")
268            # Trim the graph's node list to reachable-only.
269            self.graph.nodes[:] = [n for n in self.graph.nodes if n in reachable]
270        else:
271            self._log("[dce] no dead nodes found")
272
273        return output

280        """Run the standard optimization pipeline: fold -> fuse -> dce.
281
282        Returns the optimized output node.
283        """
284        if self.graph.output is None:
285            raise ValueError("Graph has no output node set")
286
287        folded = self.fold(self.graph.output)
288        fused = self.fuse(folded)
289        result = self.dce(fused)
290        self.graph.output = result
291        return result
